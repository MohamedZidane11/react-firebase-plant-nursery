              onError={(e) => {
                e.target.src = defaultNurseryImage;
              }}
            />
            <div className="absolute bottom-4 left-1/2 transform -translate-x-1/2 text-white text-sm bg-black bg-opacity-50 px-3 py-1 rounded-full">
              {currentImageIndex + 1} / {allImages.length}

          </div>
        </div>
      )}
    </div>
  );
};
